20  /**
21   * Checks for SNAPSHOT dependencies in the reactor, but only if requested.
22   *
23   * @author Ronald Jack Jenkins Jr.
24   */
25  @Component(role = ReleasePhase.class, hint = "conditional-check-dependency-snapshots")
26  public class ConditionalCheckDependencySnapshotsPhase extends
27      CheckDependencySnapshotsPhase {
28    @Requirement(role = AbstractMavenLifecycleParticipant.class, hint = "rtr")
29    private RTR      rtr;
30    @Requirement
31    protected Logger logger;
32  
33    /**
34     * Calls super, if permitted.
35     *
36     * @param rd
37     *          not null.
38     * @param re
39     *          not null.
40     * @param projects
41     *          not null.
42     * @return the result of the phase execution.
43     * @throws ReleaseExecutionException
44     *           as needed.
45     * @throws ReleaseFailureException
46     *           as needed.
47     */
48    @Override
49    public ReleaseResult execute(final ReleaseDescriptor rd,
50        final ReleaseEnvironment re, final List<MavenProject> projects)
51        throws ReleaseExecutionException, ReleaseFailureException {
52      final ReleaseResult result;
53      if (this.rtr.isExternalSnapshotsAllowed()) {
54        this.logger
55            .warn("External SNAPSHOT artifacts are allowed for this release. Artifacts produced by this build may not behave consistently compared to earlier builds.");
56        result = new ReleaseResult();
57        result.setResultCode(ReleaseResult.SUCCESS);
58      }
59      else {
60        result = super.execute(rd, re, projects);
61      }
62      return result;
63    }
64  
65    /**
66     * Invokes the {@link #execute(ReleaseDescriptor, ReleaseEnvironment, List)
67     * execute} method.
68     *
69     * @param rd
70     *          not null.
71     * @param re
72     *          not null.
73     * @param projects
74     *          not null.
75     * @return the result of the phase execution.
76     * @throws ReleaseExecutionException
77     *           as needed.
78     * @throws ReleaseFailureException
79     *           as needed.
80     */
81    @Override
82    public ReleaseResult simulate(final ReleaseDescriptor rd,
83        final ReleaseEnvironment re, final List<MavenProject> projects)
84        throws ReleaseExecutionException, ReleaseFailureException {
85      return this.execute(rd, re, projects);
86    }
87  }
