23  import java.util.List;
24  import java.util.Map;
25  
26  import org.apache.maven.AbstractMavenLifecycleParticipant;
27  import org.apache.maven.MavenExecutionException;
28  import org.apache.maven.execution.MavenSession;
29  import org.apache.maven.project.MavenProject;
30  import org.apache.maven.shared.release.ReleaseExecutionException;
31  import org.apache.maven.shared.release.ReleaseFailureException;
32  import org.apache.maven.shared.release.ReleaseResult;
33  import org.apache.maven.shared.release.config.ReleaseDescriptor;
34  import org.apache.maven.shared.release.env.ReleaseEnvironment;
35  import org.apache.maven.shared.release.phase.ReleasePhase;
36  import org.codehaus.plexus.component.annotations.Requirement;

38  /**
39   * Base implementation of a Smart Reactor release step. No-op unless the Smart
40   * Reactor's release property is true at runtime.
41   *
42   * @author Ronald Jack Jenkins Jr.
43   */
44  public abstract class AbstractSmartReactorReleaseStep extends
45      AbstractSmartReactorStep {
46    @Requirement(role = AbstractMavenLifecycleParticipant.class, hint = "rtr")
47    protected RTR                       rtr;
48    @Requirement(role = ReleasePhase.class)
49    protected Map<String, ReleasePhase> availablePhases;
50    @Requirement(hint = "rtr-rd")
51    protected ReleaseDescriptor         releaseDescriptor;
52    @Requirement(hint = "rtr-re")
53    protected ReleaseEnvironment        releaseEnvironment;
54  
55    /**
56     * Subclasses can override this method to configure the release descriptor
57     * injected by Plexus. The default implementation does nothing.
58     *
59     * @param session
60     *          the session to which this step applies. Not null.
61     * @param components
62     *          that this step may need. May be null.
63     */
64    protected void configureReleaseDescriptor(final MavenSession session,
65        final RTRComponents components) {}
66  
67    @Override
68    public final void execute(final MavenSession session,
69        final RTRComponents components) throws MavenExecutionException {
70      if (this.rtr.isRelease()) {
71        this.logger.info(this.getAnnouncement());
72        this.configureReleaseDescriptor(session, components);
73        this.releaseExecute(session, components);
74      }
75    }
76  
77    /**
78     * Returns the announcement that is logged when this release step begins
79     * execution.
80     *
81     * @return null or empty will prevent the corresponding log entry from
82     *         occurring.
83     */
84    protected abstract String getAnnouncement();
85  
86    /**
87     * Returns the list of phases that should be executed by this release step.
88     *
89     * @return never null or empty.
90     * @throws UnsupportedOperationException
91     *           if this step does not execute any release phases.
92     */
93    protected abstract List<String> getReleasePhases();
94  
95    /**
96     * Returns the list of phases that should be executed by this release step
97     * when rollback is required.
98     *
99     * @return never null or empty.
100    * @throws UnsupportedOperationException
101    *           if this step does not execute any release phases.
102    */
103   protected abstract List<String> getRollbackPhases();
104 
105   /**
106    * Step logic that is executed if a release was requested.
107    *
108    * @param session
109    *          the session to which this step applies. Not null.
110    * @param components
111    *          that this step may need. May be null.
112    * @throws MavenExecutionException
113    *           if any unrecoverable error occurs.
114    */
115   protected void releaseExecute(final MavenSession session,
116       final RTRComponents components) throws MavenExecutionException {
117     this.releaseEnvironment.setSettings(session.getSettings());
118     final List<MavenProject> reactor = session.getProjects();
119     // Execute the release steps.
120     try {
121       this.runPhases(reactor, this.getReleasePhases());
122     }
123     catch (final MavenExecutionException | RuntimeException e) {
124       // Rollback if ANY exception occurred, then rethrow.
125       this.logger.error("Rolling back release due to error...");
126       try {
127         this.runPhases(reactor, this.getRollbackPhases());
128       }
129       catch (final MavenExecutionException | RuntimeException e2) {
130         // Suppress this exception.
131         e.addSuppressed(e2);
132         this.logger
133             .error("Rollback unsuccessful. Check project filesystem for POM backups and other resources that must be rolled back manually.");
134       }
135       throw e;
136     }
137   }
138 
139   // Derived from DefaultReleaseManager.java in maven-release-manager, see
140   // THIRDPARTY file for further legal information.
141   private final void runPhases(final List<MavenProject> reactor,
142       final List<String> phases) throws MavenExecutionException {
143     ReleasePhase phase;
144     ReleaseResult result;
145     for (final String name : phases) {
146       phase = this.availablePhases.get(name);
147       if (phase == null) {
148         throw new SmartReactorReleaseException("Unable to find phase '" + name
149             + "' to execute");
150       }
151       try {
152         result = phase.execute(this.releaseDescriptor, this.releaseEnvironment,
153             reactor);
154       }
155       catch (final ReleaseExecutionException | ReleaseFailureException e) {
156         throw new SmartReactorReleaseException(e);
157       }
158       if (result.getResultCode() == ReleaseResult.ERROR) {
159         throw new SmartReactorReleaseException(result.getOutput());
160       }
161     }
162   }
