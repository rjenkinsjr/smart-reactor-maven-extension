16  package info.ronjenkins.maven.rtr;
17  
18  import info.ronjenkins.maven.rtr.steps.SmartReactorStep;
19  
20  import java.util.List;
21  import java.util.Map;
22  
23  import org.apache.maven.AbstractMavenLifecycleParticipant;
24  import org.apache.maven.MavenExecutionException;
25  import org.apache.maven.execution.MavenSession;
26  import org.apache.maven.project.MavenProject;
27  import org.apache.maven.project.ProjectBuilder;
28  import org.codehaus.plexus.component.annotations.Component;
29  import org.codehaus.plexus.component.annotations.Requirement;
30  import org.codehaus.plexus.logging.Logger;
31  
32  /**
33   * The entry point for the Smart Reactor Maven Extension.
34   *
35   * @author Ronald Jack Jenkins Jr.
36   */
37  @Component(role = AbstractMavenLifecycleParticipant.class, hint = "rtr")
38  public class RTR extends AbstractMavenLifecycleParticipant {
39  
40    @Requirement
41    private Logger logger;
42    @Requirement
43    private ProjectBuilder builder;
44  
45    protected List<String> startSteps;
46    protected List<String> endSuccessSteps;
47    protected List<String> endFailureSteps;
48  
49    @Requirement(role = SmartReactorStep.class)
50    protected Map<String, SmartReactorStep> availableSteps;
51  
52    private RTRComponents components;
53    private boolean disabled;
54    private boolean release;
55    private boolean backupPomsCreated;
56    private boolean externalSnapshotsAllowed;
57  
58    /**
59     * RTR entry point.
60     *
61     * @param session
62     *          the current Maven session, never null.
63     */
64    @Override
65    public void afterProjectsRead(final MavenSession session)
66        throws MavenExecutionException {
67      // Don't do anything if the Smart Reactor is disabled.
68      final MavenProject executionRoot = session.getTopLevelProject();
69      this.disabled = RTRConfig.isDisabled(session, executionRoot);
70      if (this.disabled) {
71        return;
72      }
73      this.release = RTRConfig.isRelease(session, executionRoot);
74      this.externalSnapshotsAllowed = RTRConfig.isExternalSnapshotsAllowed(
75          session, executionRoot);
76      this.logger.info("Assembling smart reactor...");
77      this.components = new RTRComponents(this.builder);
78      this.executeSteps(this.startSteps, session, this.components);
79      // Done. Maven build will proceed from here, none the wiser. ;)
80    }
81  
82    @Override
83    public void afterSessionEnd(final MavenSession session)
84        throws MavenExecutionException {
85      if (this.disabled) {
86        return;
87      }
88      if (session.getResult().hasExceptions()) {
89        this.executeSteps(this.endFailureSteps, session, this.components);
90      } else {
91        this.executeSteps(this.endSuccessSteps, session, this.components);
92      }
93    }
94  
95    private void executeSteps(final List<String> steps,
96        final MavenSession session, final RTRComponents components)
97            throws MavenExecutionException {
98      SmartReactorStep step;
99      for (final String name : steps) {
100       step = this.availableSteps.get(name);
101       if (step == null) {
102         throw new MavenExecutionException("Unable to find step '" + name
103             + "' to execute", new IllegalStateException());
104       }
105       step.execute(session, components);
106     }
107   }
108 
109   /**
110    * Indicates whether or not backup POMs were created by the release process.
111    *
112    * @return backupPomsCreated true if backup POMs have been created, false
113    *         otherwise.
114    */
115   public boolean isBackupPomsCreated() {
116     return this.backupPomsCreated;
117   }
118 
119   /**
120    * Indicates whether or not the Smart Reactor should allow a release reactor
121    * containing references to any non-reactor SNAPSHOT artifacts.
122    *
123    * @return true if allowed, false if prohibited.
124    */
125   public boolean isExternalSnapshotsAllowed() {
126     return this.externalSnapshotsAllowed;
127   }
128 
129   /**
130    * Indicates whether or not a release was requested.
131    *
132    * @return true if a release was requested, false otherwise.
133    */
134   public boolean isRelease() {
135     return this.release;
136   }
137 
138   /**
139    * Sets the flag that indicates whether or not backup POMs were created by the
140    * release process.
141    *
142    * @param backupPomsCreated
143    *          true if backup POMs have been created, false otherwise.
144    */
145   public void setBackupPomsCreated(final boolean backupPomsCreated) {
146     this.backupPomsCreated = backupPomsCreated;
147   }
148 
149 }
